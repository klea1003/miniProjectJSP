--USER 테이블 생성 
CREATE TABLE USERS (
    USERID VARCHAR2(20),
    USERPASSWORD VARCHAR2(20),
    USERNAME VARCHAR2(20),
    USERGENDER VARCHAR2(20),
    USEREMAIL VARCHAR2(20),
    ADMIN NUMBER(8) NOT NULL,
    PRIMARY KEY(USERID)    
);
COMMIT;

CREATE TABLE CART (
    CARTID NUMBER(8),
    BOOKID NUMBER(8),
    USERID VARCHAR2(20),
    BOOKAMOUNT NUMBER(8),
    ORDERCOMPLETED NUMBER(8) DEFAULT 0 NOT NULL,
    DELETECART NUMBER(8) DEFAULT 0 NOT NULL,
    ORDERID NUMBER(8) DEFAULT 0 NOT NULL,
    CARTDATE DATE,
    PRIMARY KEY(CARTID)
);
