--USER 테이블 생성 
CREATE TABLE USERS (
    USERID VARCHAR2(20),
    USERPASSWORD VARCHAR2(20),
    USERNAME VARCHAR2(20),
    USERGENDER VARCHAR2(20),
    USEREMAIL VARCHAR2(20),
    ADMIN NUMBER(8) NOT NULL,
    PRIMARY KEY(USERID)    
);
COMMIT;

--게시판 테이블 생성 
CREATE TABLE BBS (
    BBSID NUMBER(8),
    BBSTITLE VARCHAR2(100),
    USERID VARCHAR2(20),
    BBSDATE DATE,
    BBSCONTENT VARCHAR2(2048),
    BBSAVAILABLE INT,
    PRIMARY KEY(BBSID)
);
COMMIT;

--FAQ 테이블 생성 
CREATE TABLE FAQ (
    FAQID NUMBER(8),
    FAQTITLE VARCHAR2(100),
    USERID VARCHAR2(20),
    FAQDATE DATE,
    FAQCONTENT VARCHAR2(2048),
    FAQAVAILABLE INT,
    PRIMARY KEY(FAQID)
);
COMMIT;

--공지사항 테이블 생성 
CREATE TABLE NOTICE (
    NOTICEID NUMBER(8),
    NOTICETITLE VARCHAR2(100),
    USERID VARCHAR2(20),
    NOTICEDATE DATE,
    NOTICECONTENT VARCHAR2(2048),
    NOTICEAVAILABLE INT,
    PRIMARY KEY(NOTICEID)
);
COMMIT;


CREATE TABLE CART (
    CARTID NUMBER(8),
    BOOKID NUMBER(8),
    USERID VARCHAR2(20),
    BOOKAMOUNT NUMBER(8),
    ORDERCOMPLETED NUMBER(8),
    DELETECART NUMBER(8),
    CARTDATE DATE,
    PRIMARY KEY(CARTID)
);

COMMIT;
