--USER 테이블 생성 
CREATE TABLE USERS (
    USERID VARCHAR2(20),
    USERPASSWORD VARCHAR2(20),
    USERNAME VARCHAR2(20),
    USERGENDER VARCHAR2(20),
    USEREMAIL VARCHAR2(20),
    PRIMARY KEY(USERID)
);
COMMIT;

--게시판 테이블 생성 
CREATE TABLE BBS (
    BBSID NUMBER(8),
    BBSTITLE VARCHAR2(20),
    USERID VARCHAR2(20),
    BBSDATE DATE,
    BBSCONTENT VARCHAR2(2048),
    BBSAVAILABLE INT,
    PRIMARY KEY(BBSID)
);
COMMIT;

--책 테이블 생성
CREATE TABLE BOOK (
    BOOKID NUMBER(8),
    BOOKTILE VARCHAR2(20),
    BOOKPRICE NUMBER(8),
    BOOKAUTHOR VARCHAR2(20),
    BOOKCATEGORY NUMBER(8),
    BOOKPUBLISHER VARCHAR2(20),
    BOOKIMAGEPATH VARCHAR2(20),
    PRIMARY KEY(BOOKID)
);

COMMIT;


-- 장바구니 테이블 생성
CREATE TABLE CART (
    CARTID NUMBER(8),
    BOOKID NUMBER(8),
    USERID VARCHAR2(20),
    BOOKAMOUNT NUMBER(8),
    CARTDATE DATE,
    PRIMARY KEY(CARTID)
);

COMMIT;
